function ucln(a,b:int64):int64; var c:int64; begin c:=b mod a; while c<>0 do begin c:=a mod b; a:=b; b:=c; end; exit(a); end; var f:text; a,b,c,d,g,h,i:int64; begin assign(f,'frac.inp');reset(f); read(f,a,b,c,d); close(f); assign(f,'frac.out');rewrite(f); h:=b*d; a:=a*d; c:=c*b; g:=ucln(a,c); i:=ucln(g,h); g:=g div i; h:=h div i; writeln(f,g); write(f,h); close(f); end.
